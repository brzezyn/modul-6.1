SELECT ISSUES.SUMMARY, ISSUESLISTS.NAME
FROM ISSUESLISTS 
JOIN ISSUES ON ISSUES.ISSUESLIST_ID = ISSUESLISTS.ID;


SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U ON U.ID = I.USER_ID_ASSIGNEDTO;


SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_NUMBER
FROM ISSUES I
JOIN USERS U ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO;
